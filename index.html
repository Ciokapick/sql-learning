Beginner Exercises

Example Exercise 1:

Retrieve employee name, surname, country, and region:

SELECT E.FIRST_NAME, E.LAST_NAME, C.COUNTRY_NAME, R.REGION_NAME
FROM HR.EMPLOYEES E
JOIN HR.DEPARTMENTS D ON (E.DEPARTMENT_ID = D.DEPARTMENT_ID)
JOIN HR.LOCATIONS L ON (D.LOCATION_ID = L.LOCATION_ID)
JOIN HR.COUNTRIES C ON (L.COUNTRY_ID = C.COUNTRY_ID)
JOIN HR.REGIONS R ON (C.REGION_ID = R.REGION_ID);

Example Exercise 2:

Display first name for employees whose names end in 'A':

SELECT E.FIRST_NAME FROM HR.EMPLOYEES E
WHERE FIRST_NAME LIKE '%a';

Example Exercise 3:

Display employees whose first name has the second character 'a' and whose salary is under 5000:

SELECT E.FIRST_NAME FROM HR.EMPLOYEES
WHERE SALARY < 5000 AND FIRST_NAME LIKE '_a%';

Example Exercise 4:

Display first name, last name, salary, and department ID for employees whose salary is less than 3000 or greater than 4000 and they are from department 50:

SELECT FIRST_NAME, LAST_NAME, SALARY, DEPARTMENT_ID
FROM HR.EMPLOYEES
WHERE SALARY < 3000 OR (SALARY > 4000 AND DEPARTMENT_ID = 50);

Example Exercise 5:

Display the number of employees:

SELECT COUNT(*) AS EMPLOYEE_ID FROM HR.EMPLOYEES;

Example Exercise 6:

Display the total amount spent on salaries:

SELECT SUM(SALARY) FROM HR.EMPLOYEES;

Example Exercise 7:

Display job ID, job name, last name, and first name:

SELECT E.FIRST_NAME, E.LAST_NAME, E.JOB_ID, J.JOB_TITLE
FROM HR.EMPLOYEES E
JOIN HR.JOBS J ON (E.JOB_ID = J.JOB_ID);

Example Exercise 8:

Display job ID, job title, and average salary for each job:

SELECT E.JOB_ID, J.JOB_TITLE, AVG(E.SALARY)
FROM HR.JOBS J
JOIN HR.EMPLOYEES E ON J.JOB_ID = E.JOB_ID
GROUP BY E.JOB_ID, J.JOB_TITLE;

ðŸ’Ž Easy ExercisesðŸ’Ž

Exercise 1:

Employees whose salaries (including commission) are too high/too low for their job:

SELECT E.LAST_NAME, E.FIRST_NAME, J.MIN_SALARY, J.MAX_SALARY, E.SALARY, E.COMMISSION_PCT
FROM HR.EMPLOYEES E
JOIN HR.JOBS J ON E.JOB_ID = J.JOB_ID
WHERE (E.SALARY + (E.SALARY * COALESCE(E.COMMISSION_PCT, 0))) NOT BETWEEN J.MIN_SALARY AND J.MAX_SALARY;

Exercise 2:

Display the maximum salary, minimum salary, average salary, and number of employees in each department:

SELECT MIN(SALARY), MAX(SALARY), AVG(SALARY) AS MEDIA_SALARIILOR, COUNT(EMPLOYEE_ID) AS NR_ANGAJATI, DEPARTMENT_ID
FROM HR.EMPLOYEES
GROUP BY DEPARTMENT_ID;

Exercise 3:

Display employees who receive a salary higher than a value entered from the keyboard:

SELECT FIRST_NAME, LAST_NAME, SALARY
FROM HR.EMPLOYEES
WHERE SALARY > &SALARY;

Exercise 4:

Display the number of employees in each department who have a salary lower than the average salary for the company:

SELECT COUNT(*), DEPARTMENT_ID, AVG(SALARY)
FROM HR.EMPLOYEES
WHERE SALARY < (SELECT AVG(SALARY) FROM HR.EMPLOYEES)
GROUP BY DEPARTMENT_ID;

Exercise 5:

Display the number of employees in each department whose salary is within 1000 units of the average salary:

SELECT COUNT(*), DEPARTMENT_ID
FROM HR.EMPLOYEES
WHERE SALARY >= (SELECT AVG(SALARY) FROM HR.EMPLOYEES) - 1000
AND SALARY <= (SELECT AVG(SALARY) FROM HR.EMPLOYEES) + 1000
GROUP BY DEPARTMENT_ID
ORDER BY DEPARTMENT_ID ASC;

Exercise 1:

Display employees whose salary is lower than the average salary in their department:

SELECT e.EMPLOYEE_ID, e.FIRST_NAME, e.LAST_NAME, e.SALARY
  FROM HR.EMPLOYEES e
  JOIN (
      SELECT DEPARTMENT_ID, AVG(SALARY) AS AVG_SALARY
      FROM HR.EMPLOYEES
      GROUP BY DEPARTMENT_ID
  ) dept_avg ON e.DEPARTMENT_ID = dept_avg.DEPARTMENT_ID
  WHERE e.SALARY < dept_avg.AVG_SALARY;

Exercise 2:

Display the managers' departments sorted in descending order by the number of managers in each department (limited to 8 records):

SELECT E.DEPARTMENT_ID, COUNT(*) AS NR_MANAGERI
  FROM HR.EMPLOYEES
  WHERE EMPLOYEE_ID IN (SELECT MANAGER_ID FROM HR.EMPLOYEES)
  GROUP BY DEPARTMENT_ID
  ORDER BY NR_MANAGERI DESC;

Exercise 3:

Display managers' salaries:

SELECT FIRST_NAME, LAST_NAME, SALARY
  FROM HR.EMPLOYEES
  WHERE EMPLOYEE_ID IN (SELECT MANAGER_ID FROM HR.EMPLOYEES WHERE MANAGER_ID IS NOT NULL);

Exercise 4:

Display the average salaries of managers:

SELECT AVG(SALARY) AS MEDIA_SALARIILOR_MANAGERILOR
  FROM HR.EMPLOYEES
  WHERE EMPLOYEE_ID IN (SELECT MANAGER_ID FROM HR.EMPLOYEES WHERE MANAGER_ID IS NOT NULL);

Display employees who have a salary higher than at least one manager in their department:

SELECT first_name || ' ' || last_name, salary
  FROM HR.employees A
  WHERE salary > ANY (
      SELECT B.salary
      FROM HR.employees B
      WHERE A.department_id = B.department_id
      AND employee_id IN (SELECT manager_id FROM HR.employees)
  );

This command identifies the maximum salary in your dataset.

Exercise 2:

Display employees who earn a higher salary than at least one department manager in the employee's department:

SELECT first_name || ' ' || last_name, salary
  FROM HR.employees A
  WHERE salary > ANY (
      SELECT B.salary
      FROM HR.employees B, HR.departments C
      WHERE A.department_id = C.department_id
      AND employee_id IN (SELECT manager_id FROM HR.employees)
      AND A.department_id = B.department_id
  );
